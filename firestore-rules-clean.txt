rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function getUserRole() {
      return request.auth.token.role;
    }
    
    function getUserId() {
      return request.auth.uid;
    }
    
    function isMedico() {
      return getUserRole() == 'medico';
    }
    
    function isPaciente() {
      return getUserRole() == 'paciente';
    }
    
    function isAdmin() {
      return getUserRole() == 'admin';
    }
    
    match /patients/{patientId} {
      allow read, write: if isAdmin();
      
      allow read: if isMedico();
      allow write: if isMedico() && 
                     (request.resource.data.assignedDoctor == getUserId() ||
                      resource.data.assignedDoctor == getUserId());
      
      allow read: if isPaciente() && patientId == getUserId();
      allow write: if false;
      
      match /consultations/{consultationId} {
        allow read: if isAdmin() || 
                       (isMedico() && resource.data.doctorId == getUserId()) ||
                       (isPaciente() && patientId == getUserId());
        
        allow create: if isMedico() && 
                        request.resource.data.doctorId == getUserId();
        
        allow update: if isMedico() && 
                        resource.data.doctorId == getUserId();
      }
      
      match /prescriptions/{prescriptionId} {
        allow read: if isAdmin() || 
                       (isMedico() && resource.data.doctorId == getUserId()) ||
                       (isPaciente() && patientId == getUserId());
        
        allow create: if isMedico() && 
                        request.resource.data.doctorId == getUserId();
        
        allow update: if isMedico() && 
                        resource.data.doctorId == getUserId();
      }
      
      match /labResults/{labId} {
        allow read: if isAdmin() || 
                       (isMedico() && resource.data.reviewedBy == getUserId()) ||
                       (isPaciente() && patientId == getUserId());
        
        allow create: if isAdmin() || isMedico();
      }
    }
    
    match /appointments/{appointmentId} {
      allow read, write: if isAdmin();
      
      allow read: if isMedico() && resource.data.doctorId == getUserId();
      allow write: if isMedico() && 
                     request.resource.data.doctorId == getUserId();
      
      allow read: if isPaciente() && resource.data.patientId == getUserId();
      allow write: if false;
    }
    
    match /doctors/{doctorId} {
      allow read, write: if isAdmin();
      
      allow read: if isMedico() && doctorId == getUserId();
      allow write: if false;
    }
    
    match /audit/{auditId} {
      allow read: if isAdmin();
      allow write: if false;
    }
    
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
